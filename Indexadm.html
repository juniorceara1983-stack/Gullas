const SCRIPT_URL = "https://script.google.com/macros/s/AKfycbzip6qASNtkcRNBWMINpQWVoxU71AL0Ajws_BmzobGdEBA23fa0-WvqX3GjIZf_me_1/exec"; let dadosGlobais = []; // Envia nova remessa async function addEnvio() { const p = document.getElementById('prodSelect').value; const q = parseInt(document.getElementById('qtdAdd').value); if(!q) return alert("Digite uma quantidade."); await fetch(SCRIPT_URL, { method: 'POST', mode: 'no-cors', body: JSON.stringify({ action: "envio", produto: p, qtd: q }) }); alert("Envio de " + p + " registado!"); document.getElementById('qtdAdd').value = ""; carregarRelatorio(); } // Carrega e Processa o Relatório async function carregarRelatorio() { try { const response = await fetch(SCRIPT_URL); const rows = await response.json(); // rows[0] são os cabeçalhos, os dados começam em rows[1] const consolidado = {}; const cardapio = ["Massa Quente", "Massa Fria", "Massa Forno", "Bolos (inteiros)", "Suco Galao", "Empadas", "Tortas", "Pastel"]; // Inicializa o objeto cardapio.forEach(p => consolidado[p] = { envio: 0, sobra: 0 }); // Processa as linhas da planilha rows.slice(1).forEach(row => { const tipo = row[1]; // Coluna B (ENVIO ou SOBRA) const prod = row[2]; // Coluna C (Produto) const qtd = parseInt(row[3]) || 0; // Coluna D (Quantidade) if(consolidado[prod]) { if(tipo === "ENVIO") consolidado[prod].envio += qtd; if(tipo === "SOBRA") consolidado[prod].sobra += qtd; } }); exibirTabela(consolidado); } catch (e) { console.error("Erro ao processar relatório:", e); } } function exibirTabela(dados) { const tbody = document.getElementById('tableBody'); tbody.innerHTML = ""; for (const prod in dados) { const enviado = dados[prod].envio; const sobra = dados[prod].sobra; const vendido = enviado - sobra; if(enviado > 0 || sobra > 0) { tbody.innerHTML += ` ${prod} ${enviado} ${sobra} ${vendido >= 0 ? vendido : 0} `; } } } // Inicia o painel carregando os dados window.onload = () => { // Se já estiver logado, carrega if(document.getElementById('loginOverlay').style.display === 'none') { carregarRelatorio(); } };
