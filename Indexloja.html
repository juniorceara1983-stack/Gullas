const SCRIPT_URL = "https://script.google.com/macros/s/AKfycbzip6qASNtkcRNBWMINpQWVoxU71AL0Ajws_BmzobGdEBA23fa0-WvqX3GjIZf_me_1/exec"; async function enviarSobra() { const nome = document.getElementById('nomeFunc').value; const turno = document.getElementById('turnoSelect').value; const btn = document.querySelector('.btn'); if(!nome) return alert("Por favor, digite o seu nome."); btn.innerText = "A ENVIAR..."; btn.disabled = true; const inputs = document.querySelectorAll('.sobra-input'); const itens = Array.from(inputs) .filter(i => i.value !== "") // Só envia o que foi preenchido .map(i => ({ produto: i.dataset.prod, qtd: parseInt(i.value) })); if(itens.length === 0) { alert("Preencha pelo menos um item."); btn.innerText = "ENVIAR CONFERÊNCIA"; btn.disabled = false; return; } try { await fetch(SCRIPT_URL, { method: 'POST', mode: 'no-cors', body: JSON.stringify({ action: "sobra", funcionario: nome, turno: turno, itens: itens }) }); alert("Conferência enviada com sucesso!"); inputs.forEach(i => i.value = ""); } catch (e) { alert("Erro na ligação."); } finally { btn.innerText = "ENVIAR CONFERÊNCIA"; btn.disabled = false; } }
